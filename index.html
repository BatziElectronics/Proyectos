/*Con modulo rele de un canal y un sensor de movimiento (Pir)*/

int led1=13; /*Identificacion de los componentes. */

int Pir=2; /*Identificacion de los componentes. */

void setup()

{

pinMode(led1, OUTPUT); /*Verificacion si los componentes son de entrada o de salida*/

pinMode(Pir, INPUT); /*Verificacion si los componentes son de entrada o de salida*/

}

void loop() /*Parte del loop*/

{

int sensor=digitalRead(Pir); /*Asignamos un valor al sensor*/

if (sensor==HIGH) /*Si el sensor es activado*/

}

digitalWrite(Led1, HIGH); /*Enciende el módulo rele*/

}

else /*De lo contrario*/

{

digitalWrite(Led1, LOW); /*El módulo rele se apaga*/

}

}
